const fs = require("node:fs");
const fsPromises = fs.promises;
const PNG = require("pngjs").PNG;
const path = require("path");
const yauzl = require("yauzl-promise");
const { pipeline } = require("stream/promises");

async function unzip(zipFilePath, outputDir) {
  try {
    const zipfile = await yauzl.open(zipFilePath);
    await fsPromises.mkdir(outputDir, { recursive: true });

    for await (const entry of zipfile) {
      console.log("Processing entry:" + entry.filename);
      const entryPath = path.join(outputDir, entry.filename);
      console.log(`Creating file: ${entryPath}`);

      await fsPromises.mkdir(path.dirname(entryPath), { recursive: true });
      try {
        const readStream = await new Promise((resolve, reject) => {
          entry.openReadStream({}, (err, stream) => {
            if (err) {
              console.error(
                `Error opening read stream for ${entry.filename}:`,
                err
              );
              return reject(err);
            }
            resolve(stream);
          });
        });

        console.log(`Opened read stream for ${entry.filename}`);
        const writeStream = fs.createWriteStream(entryPath);

        writeStream.on("finish", () => {
          console.log(`Write operation completed for ${entryPath}`);
        });
        writeStream.on("error", (err) => {
          console.error(`Write stream error for ${entryPath}:`, err);
        });

        await pipeline(readStream, writeStream);
        console.log(`Written successfully: ${entryPath}`);
      } catch (err) {
        console.error(`Error writing ${entryPath}:`, err);
        continue;
      }
    }

    console.log("Extraction operation complete");
    await zipfile.close();
  } catch (error) {
    console.error("An error occurred during extraction:", error);
  }
}

const zipFilePath =
  "C:\\Users\\downi\\OneDrive\\Desktop\\FSWD\\Term 1\\Comp1320 (Java)\\Class 11\\Lab 11\\startingcode\\myfile.zip";
const outputDir =
  "C:\\Users\\downi\\OneDrive\\Desktop\\FSWD\\Term 1\\Comp1320 (Java)\\Class 11\\Lab 11\\startingcode\\unzipped";

unzip(zipFilePath, outputDir);
